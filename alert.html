dialog.alert("hi, 我是alert");
dialog.alert("hi, 我是alert", {
  title: "点击阴影部分关闭我",
  shadeClose: true //点击阴影可关闭
});
dialog.alert("hi, 我是alert", {
  title: "我是自定义样式的警告框哦",
  shadeClose: true, //点击阴影可关闭
  //设置标题样式，css属性名采用驼峰命名规则
  titleStyle: {
   color: "#FFFFFF",
   backgroundColor: "#4F4F4F"
  },
  //设置按钮样式，css属性名采用驼峰命名规则
  buttonStyle: {
   color: "#000000",
   border: "1px solid #DEDEDE",
   backgroundColor: "#EEEEEE"
  }
});
dialog.alert("hi, 我是alert", {
  shadeClose: true, //点击阴影可关闭
  animationType: "zoom" //zoom动画
});
