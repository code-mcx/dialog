dialog.layer();
dialog.layer({
  title: "我是不用风格的层",
  style: 2 //风格
});
dialog.layer({
  title: "我是加载了内容的层",
  //设置标题样式，css属性名采用驼峰命名规则
  titleStyle: {
   color: "#FFFFFF",
   background: "#009F95"
  },
  //html内容
  content: "hi, 我是layer层   <button id=\"btn\">点我</button>",
  //layer层添加到document后回调
  afterLoad: function(){
   document.getElementById("btn").onclick=function(){
   alert("you click me");
   };
  }
});
dialog.layer({
  title: "我是加载了内容的层",
  //html内容
  content: "hi, 我是layer层   <button id=\"btn\">点我关闭</button>",
  //layer层添加到document后回调
  afterLoad: function(){
   document.getElementById("btn").onclick=function(){
   dialog.closeLayer();
   };
  }
});
