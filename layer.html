mcxDialog.layer();


mcxDialog.layer({

  title: "我是不用风格的层",

  style: 2  //风格

});




mcxDialog.layer({

  title: "我是加载了内容的层",

  //设置标题样式，css属性名采用驼峰命名规则

  titleStyle: {

    color: "#FFFFFF",

    background: "#009F95"

  },

  //html内容

  content: "< style='padding: 18px;'>hi, 我是layer层   <button id=\"btn\">点我</button></div>",

  //layer层添加到document后回调

  afterLoad: function(){

    document.getElementById("btn").onclick=function(){

  	  alert("you click me");

    };

  }

});




mcxDialog.layer({

  title: "我是加载了内容的层",

  //html内容

  content: "< style='padding: 18px;'>hi, 我是layer层   <button id=\"btn\">点我关闭</button></div>",

  //layer层添加到document后回调

  afterLoad: function(){

    document.getElementById("btn").onclick=function(){

  	  mcxDialog.closeLayer();

    };

  }

});
